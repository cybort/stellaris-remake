namespace = zhowBattleship

# Battleships only. Chains itself.
fleet_event = {
	id = zhowBattleship.200
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_scope_valid = yes
		is_in_combat = yes
		exists = owner
		any_owned_ship = { is_ship_size = battleship } # Fleet
	}
	immediate = {
		while = {
			limit = {
				count_fleet_ships = {
					count > 0
					limit = {
						is_ship_size = battleship
						NOT = { has_ship_flag = "zhow_ignore" }
					}
				}
			}
			random_owned_ship = {
				limit = {
					is_ship_size = battleship
					NOT = { has_ship_flag = "zhow_ignore" }
				}
				zhow_set_ignore_flag = yes
			}
			owner = {
				set_variable = { which = "ZHowTmp" value = "HISBStrikeAmount" }
				divide_variable = { which = "ZHowTmp" value = 2 }
				#log = "[this.GetName]: Spawning [this.ZHowTmp] from Battleship."
				while = {
					limit = { check_variable = { which = "ZHowTmp" value > 0 } }
					if = {
						limit = { has_technology = "tech_strike_craft_2" }
							ROOT = {
								zhow_spawn_default_fighter = yes
								zhow_spawn_default_fighter = yes
								zhow_spawn_default_bomber = yes
							}
						else = {
							ROOT = {
								zhow_spawn_default_fighter = yes
								zhow_spawn_default_fighter = yes
							}
						}
					}
					subtract_variable = { which = "ZHowTmp" value = 1 }
				}
			}
		}
		fleet_event = { id = zhowStrike.210 days = 10 }
	}
}
