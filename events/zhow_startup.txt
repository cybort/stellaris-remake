namespace = zhowstart

# on game start
event = {
	id = zhowstart.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		set_global_flag = "has_zhow"

		while = {
			limit = {
				any_country = {
					NOR = {
						has_technology = tech_mass_drivers_1
						has_technology = tech_lasers_1
					}
				}
			}
			random_country = {
				limit = {
					NOR = {
						has_technology = tech_mass_drivers_1
						has_technology = tech_lasers_1
					}
				}
				random_list = {
					50 = {
						give_technology = {
							tech = tech_mass_drivers_1
							message = no
						}
					}
					50 = {
						give_technology = {
							tech = tech_lasers_1
							message = no
						}
					}
				}
			}
		}
	}
}
