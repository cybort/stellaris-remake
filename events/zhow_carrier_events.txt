namespace = zhowCarrier

# Carrier enters combat, and recurring.
fleet_event = {
	id = zhowCarrier.200
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_scope_valid = yes
		is_in_combat = yes
		exists = owner
		any_owned_ship = { is_ship_size = zhow_carrier } # Fleet
	}
	immediate = {
		owner = {
			every_owned_ship = {
				limit = {
					exists = fleet
					fleet = { is_same_value = ROOT }
					is_ship_size = zhow_carrier
				}
				if = { limit = { NOT = { has_ship_flag = "zhow_carrier_squad_1" } }
					zcarrier_launch_squadron = yes
					set_timed_ship_flag = { flag = "zhow_carrier_squad_1" days = 90 }
				}
				if = { limit = { NOT = { has_ship_flag = "zhow_carrier_squad_2" } }
					zcarrier_launch_squadron = yes
					set_timed_ship_flag = { flag = "zhow_carrier_squad_2" days = 100 }
				}
				if = { limit = { NOT = { has_ship_flag = "zhow_carrier_squad_3" } }
					zcarrier_launch_squadron = yes
					set_timed_ship_flag = { flag = "zhow_carrier_squad_3" days = 110 }
				}
			}
		}
		fleet_event = { id = zhowCarrier.200 days = 10 }
	}
}

# MTTH 30 years +/-
country_event = {
	id = zhowCarrier.100
	hide_window = yes

	trigger = {
		exists = capital_scope
		has_technology = zhow_tech_carrier_1
		has_global_flag = "zhow_enable_carrier"
	}

	mean_time_to_happen = {
		years = 29
		# If we don't have a carrier, and no project, start up a research
		# program again ASAP
		modifier = {
			factor = 0.05
			NOT = { has_special_project = zhow_carrier_project }
			count_owned_ships = {
				limit = { is_ship_size = zhow_carrier }
				count = 0
			}
		}
		modifier = {
			factor = 2
			count_owned_ships = {
				limit = { is_ship_size = zhow_carrier }
				count > 3
			}
		}
		modifier = {
			factor = 4
			count_owned_ships = {
				limit = { is_ship_size = zhow_carrier }
				count > 5
			}
		}
	}

	immediate = {
		country_event = {
			id = zhowCarrier.101
			days = 12
			random = 359
		}
	}
}

country_event = {
	id = zhowCarrier.101
	is_triggered_only = yes
	title = zhow_carrier_project
	desc = zhow_carrier_project_DESC
	picture = GFX_evt_fleet_good
	trigger = {
		exists = capital_scope
		NOT = {
			has_special_project = zhow_carrier_project
		}
	}
	immediate = {
		enable_special_project = {
			name = zhow_carrier_project
			location = capital_scope
			owner = ROOT
		}
	}
	option = {
		name = EXCELLENT
	}
}

# Triggered by special project
planet_event = {
	id = zhowCarrier.1000
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		exists = solar_system
		exists = owner
		exists = root
	}
	immediate = {
		owner = {
			country_event = {
				id = zhowCarrier.1001
			}
		}
	}
}

country_event = {
	id = zhowCarrier.1001
	is_triggered_only = yes
	title = zhow_carrier_project
	desc = zhow_carrier_completed
	picture = GFX_evt_fleet_good

	trigger = {
		exists = capital_scope
	}

	immediate = {
		create_fleet = {
			name = "Carrier Strike Group"
			effect = {
				set_owner = root
				create_ship = {
					name = random
					random_existing_design = zhow_carrier
				}
				while = {
					count = 8
					create_ship = {
						name = random
						random_existing_design = destroyer
					}
				}
				set_location = root.capital_scope
			}
		}
	}
	option = {
		name = EXCELLENT
	}
}
