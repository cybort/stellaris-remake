	fleet_event = { id = zhowStrike.100 days = 30 }

		effect = {
			set_owner = ROOT.owner

				count = 10
				zhow_spawn_fighter_hiver = yes
			}
			set_location = ROOT
			set_fleet_stance = aggressive
			set_aggro_range = 80

		}
	}
