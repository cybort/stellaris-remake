			set_location = PREVPREV

			set_aggro_range = 80

zhow_spawn_default = {
	hisb_set_timed_strike_flag = yes
	set_variable = {
		which = temp_counter
		value = HISBStrikeAmount
	}
	while = {
		limit = {
			check_variable = {
				which = temp_counter
				value > 0
			}
		}
		if = {
			limit = {
				has_technology = tech_strike_craft_2
			}
			prev = {
				fleet = {
					hisb_create_random_craft = yes
				}
			}
			else = {
				prev = {
					fleet = {
						hisb_create_fighter_craft = yes
					}
				}
			}
		}
		change_variable = {
			which = temp_counter
			value = -1
		}
	}
}
